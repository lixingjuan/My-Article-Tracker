class MyPromise {
  constructor() {
    console.log("2");
  }

  all(args) {
    if (!Array.isArray(args)) {
      throw TypeError("params must be array");
    }
    const fns = args.map(Promise.resolve);
    const result = [];
    let counter = 0;

    return new Promise((resolve, reject) => {
      fns.forEach((fn, index) => {
        fn.then(
          (res) => {
            result[index] = res;
            counter += 1;

            if (counter === fn.length) {
              resolve(result);
            }
          },
          (err) => {
            reject(err);
          }
        );
      });
    });
  }
}
